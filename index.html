// --- NOVO MOTOR DE RECONHECIMENTO LÉO KASAMA --- function processarAPI(dados) { const categorias = { animes: [], filmes: [], series: [], noticias: [], tv_aberta: [] }; dados.forEach(item => { const nome = item.name.toUpperCase(); // Lógica de Reconhecimento por Palavras-Chave if (nome.includes("ANIME") || nome.includes("NARUTO") || nome.includes("OTAKU")) { categorias.animes.push(item); } else if (nome.includes("GLOBO") || nome.includes("SBT") || nome.includes("RECORD") || nome.includes("BAND")) { categorias.tv_aberta.push(item); } else if (nome.includes("NEWS") || nome.includes("NOTICIA") || nome.includes("CNN")) { categorias.noticias.push(item); } else if (item.is_series) { categorias.series.push(item); } else { categorias.filmes.push(item); } }); console.log("Sistema Uberlândia: Conteúdo Organizado!", categorias); exibirNoPainel(categorias); }
